## Descripcion

Implementar el usecase `ChatWithDocument` que permite al coordinador conversar con Alizia para modificar un documento de coordinacion mediante lenguaje natural. El historial se persiste en backend (P15) con auto-compactacion. Los tools disponibles se filtran segun el estado del documento (P12).

```go
func (uc *ChatWithDocumentUsecase) Execute(
    ctx context.Context,
    orgID uint,
    docID uint,
    message string,
) (*ChatResponse, error) {

    // 1. Cargar documento completo
    doc, err := uc.docRepo.GetWithPreloads(ctx, orgID, docID)
    if err != nil {
        return nil, err
    }

    // Auto-transicion pending → in_progress (P1)
    if doc.Status == "pending" {
        uc.docRepo.UpdateStatus(ctx, orgID, docID, "in_progress")
        doc.Status = "in_progress"
    }

    // Validar estado editable (in_progress o published)
    if doc.Status != "in_progress" && doc.Status != "published" {
        return nil, NewForbiddenError("solo se puede chatear con documentos en edicion o publicados")
    }

    // 2. Persistir mensaje del usuario
    uc.chatRepo.SaveMessage(ctx, &entities.CoordDocChatMessage{
        CoordinationDocumentID: docID,
        Role:    "user",
        Content: message,
    })

    // 3. Verificar si necesita auto-compactacion (P15)
    msgCount, _ := uc.chatRepo.GetMessageCount(ctx, docID)
    if msgCount > uc.config.ChatCompactThreshold { // default: 50
        uc.compactHistory(ctx, docID)
    }

    // 4. Cargar historial desde backend
    history, _ := uc.chatRepo.GetHistory(ctx, docID, 0, 0) // todo el historial

    // 5. Construir system prompt con contexto del documento
    systemPrompt := buildChatSystemPrompt(doc)

    // 6. Construir mensajes: system + history + nuevo mensaje
    messages := []openai.ChatCompletionMessage{
        {Role: "system", Content: systemPrompt},
    }
    for _, h := range history {
        messages = append(messages, openai.ChatCompletionMessage{
            Role:    h.Role,
            Content: h.Content,
        })
    }

    // 7. Seleccionar tools segun estado (P12)
    tools := getToolsForStatus(doc.Status)

    // 8. Enviar a Azure OpenAI con tools y timeout reducido (P17)
    resp, err := uc.aiClient.CreateChatCompletion(ctx, openai.ChatCompletionRequest{
        Model:                "gpt-5-mini",
        Messages:             messages,
        Tools:                tools,
        MaxCompletionTokens:  4096,
    })
    if err != nil {
        // Reintento 1 vez para errores retriables (P17)
        if isRetriableError(err) {
            resp, err = uc.aiClient.CreateChatCompletion(ctx, /* same request */)
            if err != nil {
                return nil, NewServiceUnavailableError("IA no disponible, intenta de nuevo")
            }
        } else {
            return nil, err
        }
    }

    // 9. Procesar tool_calls si existen (best-effort + retry P18)
    actions := []ActionResult{}
    choice := resp.Choices[0]

    if len(choice.Message.ToolCalls) > 0 {
        for _, toolCall := range choice.Message.ToolCalls {
            result, err := uc.executeToolCall(ctx, doc, toolCall)
            if err != nil {
                // Retry 1 vez (P18)
                result, err = uc.executeToolCall(ctx, doc, toolCall)
                if err != nil {
                    actions = append(actions, ActionResult{
                        Tool:    toolCall.Function.Name,
                        Result:  fmt.Sprintf("Error: %s", err.Error()),
                        Success: false,
                    })
                    continue
                }
            }
            actions = append(actions, ActionResult{
                Tool:    toolCall.Function.Name,
                Result:  result,
                Success: true,
            })
        }
    }

    // 10. Persistir respuesta del assistant
    uc.chatRepo.SaveMessage(ctx, &entities.CoordDocChatMessage{
        CoordinationDocumentID: docID,
        Role:      "assistant",
        Content:   choice.Message.Content,
        ToolCalls: serializeToolCalls(choice.Message.ToolCalls),
    })

    // 11. Retornar respuesta
    return &ChatResponse{
        Message: choice.Message.Content,
        Actions: actions,
    }, nil
}
```

Cuando el historial supera el umbral (configurable, default 50 mensajes):

```go
func (uc *ChatWithDocumentUsecase) compactHistory(ctx context.Context, docID uint) {
    history, _ := uc.chatRepo.GetHistory(ctx, docID, 0, 0)

    // Armar resumen con el LLM
    summaryPrompt := "Resumi esta conversacion en un parrafo, " +
        "destacando las acciones realizadas y el estado actual del documento."

    messages := []openai.ChatCompletionMessage{
        {Role: "system", Content: summaryPrompt},
    }
    for _, h := range history {
        messages = append(messages, openai.ChatCompletionMessage{
            Role: h.Role, Content: h.Content,
        })
    }

    resp, err := uc.aiClient.CreateChatCompletion(ctx, openai.ChatCompletionRequest{
        Model:    "gpt-5-mini",
        Messages: messages,
    })
    if err != nil {
        return // si falla la compactacion, seguir sin compactar
    }

    summary := resp.Choices[0].Message.Content
    uc.chatRepo.CompactHistory(ctx, docID, summary)
}
```

### System prompt

```go
func buildChatSystemPrompt(doc *CoordinationDocument) string {
    var sb strings.Builder

    sb.WriteString("Eres Alizia, asistente pedagogica de coordinacion educativa. ")
    sb.WriteString("Ayudas a coordinadores a modificar documentos de coordinacion.\n\n")

    sb.WriteString("DOCUMENTO ACTUAL:\n")
    sb.WriteString(fmt.Sprintf("- Titulo: %s\n", doc.Name))
    sb.WriteString(fmt.Sprintf("- Area: %s\n", doc.Area.Name))
    sb.WriteString(fmt.Sprintf("- Estado: %s\n\n", doc.Status))

    // Instrucciones segun estado (P12)
    if doc.Status == "published" {
        sb.WriteString("⚠️ IMPORTANTE: El documento esta publicado. Las clases son INMUTABLES.\n")
        sb.WriteString("Solo podes editar secciones narrativas. Si el coordinador te pide ")
        sb.WriteString("editar una clase, explicale que las clases no se pueden modificar ")
        sb.WriteString("en un documento publicado y que el teacher puede editarlas.\n\n")
    }

    // Secciones
    sb.WriteString("SECCIONES:\n")
    for key, value := range doc.Sections {
        sb.WriteString(fmt.Sprintf("- %s (key: %s): %s\n", key, key, truncate(value, 500)))
    }

    // Disciplinas y clases
    sb.WriteString("\nMATERIAS Y CLASES:\n")
    for _, subject := range doc.Subjects {
        sb.WriteString(fmt.Sprintf("\n## %s\n", subject.Subject.Name))
        sb.WriteString("Topics asignados:\n")
        for _, topic := range subject.SubjectTopics {
            sb.WriteString(fmt.Sprintf("  - ID:%d %s\n", topic.TopicID, topic.Topic.Name))
        }
        sb.WriteString("Clases:\n")
        for _, class := range subject.Classes {
            topicIDs := extractTopicIDs(class.ClassTopics)
            sb.WriteString(fmt.Sprintf("  - ID:%d Clase %d: %s (topics: %v)\n",
                class.ID, class.ClassNumber, class.Title, topicIDs))
        }
    }

    sb.WriteString("\nUSA LAS HERRAMIENTAS DISPONIBLES para realizar cambios. ")
    sb.WriteString("Siempre confirma al usuario que cambios realizaste.")

    return sb.String()
}
```

### Ejecucion de tool calls

```go
func (uc *ChatWithDocumentUsecase) executeToolCall(
    ctx context.Context,
    doc *CoordinationDocument,
    toolCall openai.ToolCall,
) (string, error) {
    var args map[string]interface{}
    json.Unmarshal([]byte(toolCall.Function.Arguments), &args)

    switch toolCall.Function.Name {
    case "update_section":
        key := args["section_key"].(string)
        content := args["content"].(string)
        err := uc.updateSectionsUsecase.Execute(ctx, doc.ID, key, content)
        if err != nil {
            return "", err
        }
        return fmt.Sprintf("Seccion '%s' actualizada", key), nil

    case "append_to_section":
        key := args["section_key"].(string)
        content := args["content"].(string)
        err := uc.appendSectionUsecase.Execute(ctx, doc.ID, key, content)
        if err != nil {
            return "", err
        }
        return fmt.Sprintf("Contenido agregado a seccion '%s'", key), nil

    case "update_class":
        classID := uint(args["class_id"].(float64))
        update := entities.CoordDocClass{ID: classID}
        if title, ok := args["title"].(string); ok {
            update.Title = title
        }
        if objective, ok := args["objective"].(string); ok {
            update.Objective = objective
        }
        err := uc.updateClassUsecase.Execute(ctx, &update)
        if err != nil {
            return "", err
        }
        if topicIDs, ok := args["topic_ids"]; ok {
            ids := toUintSlice(topicIDs)
            uc.updateClassTopicsUsecase.Execute(ctx, classID, ids)
        }
        return fmt.Sprintf("Clase %d actualizada", classID), nil

    default:
        return "", fmt.Errorf("tool desconocida: %s", toolCall.Function.Name)
    }
}
```

```go
type ChatResponse struct {
    Message string         `json:"message"`
    Actions []ActionResult `json:"actions"`
}

type ActionResult struct {
    Tool    string `json:"tool"`
    Result  string `json:"result"`
    Success bool   `json:"success"`
}
```

## Notas

- El historial ya NO se envia desde el frontend — se carga desde `coord_doc_chat_messages` (P15)
- El request del chat es solo `{ "message": "..." }` — sin history
- Auto-compactacion silenciosa: si falla, el chat sigue funcionando sin compactar
- El campo `Success` en `ActionResult` permite al frontend distinguir acciones exitosas de fallidas
- Best-effort: si un tool falla despues de 1 retry, se reporta pero los demas se ejecutan (P18)
- Timeout reducido para llamadas a la IA (configurable, default 45s) (P17)
- Chat funciona en `in_progress` Y `published`, pero con tools filtrados (P12)
