## Descripción

Implementar los repositorios GORM que satisfacen `CoordinationProvider`, `CoordinationChatProvider` y `CoordinationSnapshotProvider`. La complejidad principal está en los preloads anidados para `GetByID` (5 niveles de profundidad) y en las transacciones para `Create` (múltiples tablas en una operación atómica).

### CoordinationRepository

```go
package coordination

import (
    "context"

    "go-alizia/src/core/entities"
    "go-alizia/src/core/providers"
    "gorm.io/gorm"
)

type coordinationRepository struct {
    db *gorm.DB
}

func NewCoordinationRepository(db *gorm.DB) providers.CoordinationProvider {
    return &coordinationRepository{db: db}
}

// GetByID con preloads completos del arbol
func (r *coordinationRepository) GetByID(ctx context.Context, orgID, docID uint) (*entities.CoordinationDocument, error) {
    var doc entities.CoordinationDocument
    err := r.db.WithContext(ctx).
        Where("id = ? AND organization_id = ?", docID, orgID).
        Preload("Area").
        Preload("DocTopics.Topic").
        Preload("Subjects.Subject").
        Preload("Subjects.SubjectTopics.Topic").
        Preload("Subjects.Classes", func(db *gorm.DB) *gorm.DB {
            return db.Order("class_number ASC")
        }).
        Preload("Subjects.Classes.ClassTopics.Topic").
        First(&doc).Error
    if err != nil {
        return nil, err
    }
    return &doc, nil
}

// List sin preloads profundos (solo resumen)
func (r *coordinationRepository) List(ctx context.Context, orgID uint, areaID *uint) ([]entities.CoordinationDocument, error) {
    var docs []entities.CoordinationDocument
    query := r.db.WithContext(ctx).
        Where("organization_id = ?", orgID).
        Preload("Area")
    if areaID != nil {
        query = query.Where("area_id = ?", *areaID)
    }
    err := query.Find(&docs).Error
    return docs, err
}

// Create con transaccion (documento + junction tables)
func (r *coordinationRepository) Create(ctx context.Context, doc *entities.CoordinationDocument) error {
    return r.db.WithContext(ctx).Transaction(func(tx *gorm.DB) error {
        if err := tx.Create(doc).Error; err != nil {
            return err
        }
        return nil
    })
}

// UpdateStatus para transiciones de estado
func (r *coordinationRepository) UpdateStatus(ctx context.Context, orgID, docID uint, status string) error {
    return r.db.WithContext(ctx).
        Model(&entities.CoordinationDocument{}).
        Where("id = ? AND organization_id = ?", docID, orgID).
        Update("status", status).Error
}

// Delete con cascade (Postgres maneja el cascade via FKs)
func (r *coordinationRepository) Delete(ctx context.Context, orgID, docID uint) error {
    return r.db.WithContext(ctx).
        Where("id = ? AND organization_id = ?", docID, orgID).
        Delete(&entities.CoordinationDocument{}).Error
}

// GetUnassignedSubtopics para validacion al publicar (P13)
func (r *coordinationRepository) GetUnassignedSubtopics(ctx context.Context, docID uint) ([]entities.Topic, error) {
    var topics []entities.Topic
    err := r.db.WithContext(ctx).Raw(`
        SELECT t.id, t.name
        FROM coord_doc_topics cdt
        JOIN topics t ON t.id = cdt.topic_id
        WHERE cdt.coordination_document_id = ?
          AND cdt.topic_id NOT IN (
            SELECT cdst.topic_id
            FROM coord_doc_subject_topics cdst
            JOIN coordination_document_subjects cds ON cds.id = cdst.coord_doc_subject_id
            WHERE cds.coordination_document_id = ?
          )
    `, docID, docID).Scan(&topics).Error
    return topics, err
}
```

### ChatRepository (P15)

```go
type chatRepository struct {
    db *gorm.DB
}

func NewChatRepository(db *gorm.DB) providers.CoordinationChatProvider {
    return &chatRepository{db: db}
}

func (r *chatRepository) SaveMessage(ctx context.Context, msg *entities.CoordDocChatMessage) error {
    return r.db.WithContext(ctx).Create(msg).Error
}

func (r *chatRepository) GetHistory(ctx context.Context, docID uint, limit, offset int) ([]entities.CoordDocChatMessage, error) {
    var messages []entities.CoordDocChatMessage
    query := r.db.WithContext(ctx).
        Where("coordination_document_id = ?", docID).
        Order("created_at ASC")
    if limit > 0 {
        query = query.Limit(limit).Offset(offset)
    }
    err := query.Find(&messages).Error
    return messages, err
}

func (r *chatRepository) GetMessageCount(ctx context.Context, docID uint) (int64, error) {
    var count int64
    err := r.db.WithContext(ctx).
        Model(&entities.CoordDocChatMessage{}).
        Where("coordination_document_id = ?", docID).
        Count(&count).Error
    return count, err
}

// CompactHistory: reemplaza todo el historial por un resumen system
func (r *chatRepository) CompactHistory(ctx context.Context, docID uint, summary string) error {
    return r.db.WithContext(ctx).Transaction(func(tx *gorm.DB) error {
        // Borrar historial existente
        if err := tx.Where("coordination_document_id = ?", docID).
            Delete(&entities.CoordDocChatMessage{}).Error; err != nil {
            return err
        }
        // Insertar resumen como mensaje system
        return tx.Create(&entities.CoordDocChatMessage{
            CoordinationDocumentID: docID,
            Role:    "system",
            Content: summary,
        }).Error
    })
}
```

### SnapshotRepository (P20)

```go
type snapshotRepository struct {
    db *gorm.DB
}

func NewSnapshotRepository(db *gorm.DB) providers.CoordinationSnapshotProvider {
    return &snapshotRepository{db: db}
}

func (r *snapshotRepository) CreateSnapshot(ctx context.Context, snapshot *entities.CoordDocSnapshot) error {
    return r.db.WithContext(ctx).Create(snapshot).Error
}

func (r *snapshotRepository) GetSnapshots(ctx context.Context, docID uint) ([]entities.CoordDocSnapshot, error) {
    var snapshots []entities.CoordDocSnapshot
    err := r.db.WithContext(ctx).
        Where("coordination_document_id = ?", docID).
        Order("created_at DESC").
        Find(&snapshots).Error
    return snapshots, err
}

func (r *snapshotRepository) CountSnapshots(ctx context.Context, docID uint) (int64, error) {
    var count int64
    err := r.db.WithContext(ctx).
        Model(&entities.CoordDocSnapshot{}).
        Where("coordination_document_id = ?", docID).
        Count(&count).Error
    return count, err
}

// DeleteOldestSnapshot: elimina el snapshot mas viejo para mantener el limite de 3
func (r *snapshotRepository) DeleteOldestSnapshot(ctx context.Context, docID uint) error {
    return r.db.WithContext(ctx).
        Where("id = (?)",
            r.db.Model(&entities.CoordDocSnapshot{}).
                Where("coordination_document_id = ?", docID).
                Order("created_at ASC").
                Limit(1).
                Select("id"),
        ).
        Delete(&entities.CoordDocSnapshot{}).Error
}

// RestoreSnapshot: aplica sections y classes del snapshot al documento
func (r *snapshotRepository) RestoreSnapshot(ctx context.Context, docID, snapshotID uint) error {
    var snapshot entities.CoordDocSnapshot
    if err := r.db.WithContext(ctx).
        Where("id = ? AND coordination_document_id = ?", snapshotID, docID).
        First(&snapshot).Error; err != nil {
        return err
    }

    return r.db.WithContext(ctx).Transaction(func(tx *gorm.DB) error {
        // Actualizar sections del documento
        if err := tx.Model(&entities.CoordinationDocument{}).
            Where("id = ?", docID).
            Update("sections", snapshot.Sections).Error; err != nil {
            return err
        }

        // Restaurar classes: borrar existentes y recrear desde snapshot
        // La logica de deserializacion de classes JSONB va aqui
        return nil
    })
}
```

## Notas

- `GetByID` necesita 5+ niveles de `Preload` para traer el arbol completo
- `List` solo hace `Preload("Area")` para mantener la respuesta liviana
- `Create` usa transaccion; GORM puede crear relaciones anidadas automaticamente
- `Delete` depende del `ON DELETE CASCADE` de Postgres — incluye chat y snapshots
- `UpdateStatus` es un update parcial para transiciones de estado (P1, P2)
- `GetUnassignedSubtopics` usa raw SQL para la validacion al publicar (P13)
- `CompactHistory` es destructivo: borra todo y deja solo el resumen (P15)
- `DeleteOldestSnapshot` mantiene el limite de 3 snapshots (P20) — se llama antes de crear uno nuevo si ya hay 3
- `RestoreSnapshot` necesita logica adicional para deserializar el JSONB de classes y recrear los registros en las tablas de clases
- Considerar agregar `Order("class_number ASC")` en el preload de `Classes` para mantener el orden
