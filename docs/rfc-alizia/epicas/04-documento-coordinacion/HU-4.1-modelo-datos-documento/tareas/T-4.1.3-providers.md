```go
    SetSubjects(ctx context.Context, docID uint, subjects []entities.CoordinationDocumentSubject) error

// P15: Chat messages
type CoordinationChatProvider interface {

    GetHistory(ctx context.Context, docID uint, limit, offset int) ([]entities.CoordDocChatMessage, error)
    GetMessageCount(ctx context.Context, docID uint) (int64, error)
    CompactHistory(ctx context.Context, docID uint, summary string) error
}

// P20: Snapshots
type CoordinationSnapshotProvider interface {
    CreateSnapshot(ctx context.Context, snapshot *entities.CoordDocSnapshot) error

    DeleteOldestSnapshot(ctx context.Context, docID uint) error
    CountSnapshots(ctx context.Context, docID uint) (int64, error)
}
```
